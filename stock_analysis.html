import asyncio import json import pandas as pd from datetime import datetime, timedelta import baostock as bs # 全局变量存储数据 stock_data = None stock_code = None def get_stock_data(code): global stock_data, stock_code stock_code = code # 计算一年前的日期 end_date = datetime.now().strftime('%Y-%m-%d') start_date = (datetime.now() - timedelta(days=365)).strftime('%Y-%m-%d') try: # 登录Baostock lg = bs.login() if lg.error_code != '0': return f"错误：{lg.error_msg}" # 使用Baostock获取K线数据 rs = bs.query_history_k_data_plus( code=f"sh.{code}" if code.startswith('6') else f"sz.{code}", fields="date,open,high,low,close,volume", start_date=start_date, end_date=end_date, frequency="d", adjustflag="2" # 前复权 ) # 处理数据 data_list = [] while (rs.error_code == '0') & rs.next(): data_list.append(rs.get_row_data()) # 登出Baostock bs.logout() # 转换为DataFrame stock_zh_a_hist_df = pd.DataFrame( data_list, columns=['日期', '开盘', '最高', '最低', '收盘', '成交量'] ) # 转换数据类型 stock_zh_a_hist_df['开盘'] = pd.to_numeric(stock_zh_a_hist_df['开盘']) stock_zh_a_hist_df['最高'] = pd.to_numeric(stock_zh_a_hist_df['最高']) stock_zh_a_hist_df['最低'] = pd.to_numeric(stock_zh_a_hist_df['最低']) stock_zh_a_hist_df['收盘'] = pd.to_numeric(stock_zh_a_hist_df['收盘']) stock_zh_a_hist_df['成交量'] = pd.to_numeric(stock_zh_a_hist_df['成交量']) stock_data = stock_zh_a_hist_df return stock_zh_a_hist_df except Exception as e: return f"错误：{str(e)}" def prepare_chart_data(df): # 准备ECharts K线图数据 dates = df['日期'].tolist() data = [] for i, date in enumerate(dates): data.append([ date, float(df.iloc[i]['开盘']), float(df.iloc[i]['收盘']), float(df.iloc[i]['最低']), float(df.iloc[i]['最高']), float(df.iloc[i]['成交量']) ]) return data def update_chart(): if stock_data is not None: chart_data = prepare_chart_data(stock_data) # 调用JavaScript更新图表 js_code = f"updateChart({json.dumps(chart_data)})" pyscript.run_js(js_code) def download_csv(): if stock_data is not None: # 转换为CSV字符串 csv_content = stock_data.to_csv(index=False, encoding='utf-8-sig') # 调用JavaScript下载 js_code = f"downloadCSV({json.dumps(csv_content)}, '{stock_code}.csv')" pyscript.run_js(js_code) # 绑定查询按钮事件 async def on_query(): code = pyscript.document.getElementById('stock-code').value.strip() error_div = pyscript.document.getElementById('error-message') if not code: error_div.innerText = "请输入股票代码" return # 清空错误信息 error_div.innerText = "" # 获取数据 result = get_stock_data(code) if isinstance(result, str) and result.startswith("错误"): error_div.innerText = result else: # 更新图表 update_chart() # 绑定下载按钮事件 async def on_download(): if stock_data is None: error_div = pyscript.document.getElementById('error-message') error_div.innerText = "请先查询股票数据" return download_csv() # 注册事件监听器 from pyscript import when @when('click', '#query-btn') async def on_query_handler(): await on_query() @when('click', '#download-btn') async def on_download_handler(): await on_download()
